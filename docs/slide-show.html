Options

{
    effect_type          : 'slide',  // 'scroll' or 'slide' or 'fade'
    duration_time        : 8,        // [second]
    transition_time      : 1,        // [second]
    is_background_visible: true,     // true or false
    is_picture_scroll    : false,    // true or false
	is_random_timing     : true,     // true or false
    is_side_slide_visible: false,    // true or false
    background_opacity   : 0.33,     // 0.0 - 1.0
    zoom_rate            : 1.05,     // Zoom rate of Ken Burns effect
}
